الـقـنـاة الـبـوت مـشـاركـة

- حـول الـمـوقـع

سـبـب الإنـشـاء – قـمـت بـعـمـل الـمـوقـع لـتـسـهـيـل الـوصـول الـى جـمـيـع مـنـشـورات الـقـنـاة بـسـهـولـة.

الـقـنـاة – قـنـاتـي خـاصـة ولـكـن تـسـتـطـيـع الـوصـول لـهـا عـن طـريـق قـنـاة @Rand2m سـتـجـد رابـط الـدعـوة هـنـاك

الـبـوتـات – يـوجـد اكـثـر مـن بـوت فـي قـنـاتـي اغـلـب هـذه الـبـوتـات تـخـص الـقـنـاة والـبـوتـات الـتي تـخـص الـمـشـتـركـيـن مـوجـودة فـي قـسـم الـعـشـوائـيـات.

الـمـواقـع – جـمـيـع الـمـواقـع تـخـص قـنـاتـي وهـنـاك بـعـض الـمـواقـع الـتـي لا تـنـتـمـي الـى الـقـنـاة مـوجـودة بـقـسـم الـعـشـوائـيـات.

الادوات – الأدوات الـتـي لا تـخـص الـقـنـاة مـوجـودة فـي قـسـم الـعـشـوائـيـات.

الـمـلـفـات – الـمـلـفـات الـتـي لا تـنـتـمـي الـى الـقـنـاة سـتـجـد بـداخـلـهـا حـقـوق الـمـطور، لـيـسـت جـمـيـع الـمـلـفـات تـنـتـمـي لـلـقـنـاة.

الـتـطـبـيـقـات/strong> – الـكـثـيـر مـن الـتـطبـيـقـات لا تـنـتـمـي لـلـقـنـاة وقـد تـم نـشـرهـم بـغـرض الـمـسـاعـدة لا اكـثـر، اغـلب الـتـطـبـيـقـات قـد تـم تـعـديـلـهـا مـن قِـبـل الـمـطـور.

مُـهـم! – أرجـوا ان تـشـتـركـوا بـقـنـاتـي سـتـجـدون هـنـاك جـمـيـع مـاتـحـتـاجـونـه مـن ادوات ومـواقـع وبـوتـات ومـلـفـات وتـطـبـيـقات وسـورسـات وجـمـيـع مـا سـتـحـتـاجـة فـي مـسـيـرتـك الـبـرمـجـيـة.
